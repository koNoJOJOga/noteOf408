```c
int firstMissingNum(SqList& L) {
    int n=L.length;
    int nums[]=(int*)mallloc(sizeof(int)*(n+1));
    int i=0;
    while(i<=n) {   // 初始化数组
        nums[i++]=0
    }
    for(i=0; i<0; i++) {    // 建立映射
        if(L.data[i]>0) {
            nums[L.data[i]]=1;
        }
    }
    for(i=1; i<=n; i++) {   // 寻找缺失的第一个正整数
        if(nums[i]==0) break;
    }
    return i;
}
```
